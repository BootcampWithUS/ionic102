            .controller('LoadingCtrl', function($scope, $ionicLoading) {
              $scope.show = function() {
                $ionicLoading.show({
                  template: 'Loading...'
                });
              };
              $scope.hide = function(){
                $ionicLoading.hide();
              };
            });
